## Criar e ativar o venv
- python3 -m venv .venv
- source .venv/bin/activate
## Instalar as dependencias
- pip install -r requirements.txt

## Configurar chave de api e git-ignore
- echo "GOOGLE_API_KEY=|||CHAVE_AQUI|||" > .env
- echo ".env" > .gitignore
- echo ".venv" >> .gitignore
- echo ".env.swp" >> .gitignore
